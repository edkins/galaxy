fn myprint(x)

    z = x.srlw(4)
    z = z.and(15)
    y = y.zxbw()
    z = z.zxbw()
    y = y.sllw(8)
    y = y.or(z)
    y = y.addb(48)
    z = y.cmpgtb(57)
    z = z.and(39)

    write y
